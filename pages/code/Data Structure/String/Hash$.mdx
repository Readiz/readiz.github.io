## Hash

struct Node {
    int val;
};

struct Vector {
    Node *data;
    int capa;
    int size;

    Vector() {
        capa = 1;
        size = 0;
        data = new Node[capa];
    }
};

struct Hash {
    Vector data[DATA_MAX];
};
